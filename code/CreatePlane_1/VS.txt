// 정점쉐이더 반환값이 무조건 float4 형 이어야 한다.
// 시멘틱(의미구조) : SV_POSITION, POSITION) 
struct VS_Out
{
	float4 p : SV_POSITION;// 반드시 float4
	float4 c  :COLOR0;
};
VS_Out VSMain( float2 p : POS, float4 c:COLOR) 
{
  VS_Out  vsOut=(VS_Out)0;
  vsOut.p = float4(p.x, p.y, 0, 1);
  vsOut.c = c;
  return vsOut;
}
